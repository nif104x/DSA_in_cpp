// question link: https://leetcode.com/problems/remove-nth-node-from-end-of-list


class Solution {
public:
    ListNode* removeNthFromEnd(ListNode* head, int n) {
        int count = 0;
        ListNode *temp = head;
        while(temp){
            count++;
            temp = temp->next;
        }

        count-=n;
        
        if(count==0){  // means deleting the first Node from the first 5-5=0 (edge case)
            ListNode *del = head;
            head = head->next;
            delete del;
            return head;
        }

        ListNode *curr = head, *prev = NULL;

        while(count--){
            prev = curr;
            curr = curr->next;
        }
        prev->next = curr->next;
        delete curr;
        return head;
    }
};
